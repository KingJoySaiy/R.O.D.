* 简单输入输出外挂
```c++
LL read() {                 //输入非负整数

    LL x = 0;
    char ch = getchar();
    while (!isdigit(ch)) ch = getchar();
    while (isdigit(ch)) x = (x << 1) + (x << 3) + ch - 48, ch = getchar();
    return x;
}

template<class T>
inline void out(T x) {      //输出非负整数，负数用printf吧

    if (x > 9) out(x / 10);
    putchar(x % 10 + '0');
}
```

* 输入外挂进阶版
```c++
template <class T>
bool readLong(T &ret) {     //输入short, int, long, long long

    char c;
    int sgn;
    if ((c = getchar()) == EOF) return false;
    while (c != '-' and (c < '0' || c > '9')) c = getchar();
    sgn = (c == '-' ? -1 : 1);
    ret = (c == '-') ? 0 : (c - '0');
    while (c = getchar(), c >= '0' and c <= '9') ret = ret * 10 + (c - '0');
    ret *= sgn;
    return true;
}

template <class T>
bool readDouble(T &num) {   //输入float, double

    char in = getchar();
    double Dec = 0.1;
    bool IsN = false, IsD = false;
    if (in == EOF) return false;
    while (in != '-' and in != '.' and (in < '0' || in > '9')) in = getchar();
    if (in == '-') {
        IsN = true;
        num = 0;
    } else if (in == '.') {
        IsD = true;
        num = 0;
    } else num = in - '0';
    if (!IsD) {
        while (in = getchar(), in >= '0' and in <= '9') {
            num *= 10;
            num += in - '0';
        }
    }
    if (in != '.') {
        if (IsN) num = -num;
        return true;
    } else {
        while (in = getchar(), in >= '0' and in <= '9') {
            num += Dec * (in - '0');
            Dec *= 0.1;
        }
    }
    if (IsN) num = -num;
    return true;
}
```
